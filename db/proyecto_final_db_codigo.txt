create schema empresa

create table empresa.PIEZAS (CODIGO VARCHAR(6) not null,nombre VARCHAR(20),PRECIO FLOAT NOT NULL,DESCRIPCION TEXT,primary key (CODIGO));
create table empresa.PROVEEDORES (CODIGO VARCHAR(6) not null,nombre VARCHAR(20),APELLIDO VARCHAR(30) NOT NULL,DIRECCION VARCHAR(40),primary key (CODIGO));
create table empresa.PROYECTOS (CODIGO VARCHAR(6) not null,nombre VARCHAR(20),CIUDAD VARCHAR(40),primary key (CODIGO));


create table EMPRESA.GESTION (CODPROVEEDOR VARCHAR(6) NOT NULL, CODPIEZA  VARCHAR(6) NOT NULL,CODPROYECTO VARCHAR(6) NOT NULL,CANTIDAD NUMERIC,PRIMARY KEY(CODPROVEEDOR, CODPIEZA, CODPROYECTO),FOREIGN KEY (CODPROVEEDOR) REFERENCES PROVEEDORES (CODIGO) ON DELETE CASCADE,FOREIGN KEY (CODPIEZA) REFERENCES PIEZAS (CODIGO) ON DELeTE CASCADE,FOREIGN KEY (CODPROYECTO) REFERENCES PPROYECTOS (CODIGO) ON DELeTE CASCADE);
